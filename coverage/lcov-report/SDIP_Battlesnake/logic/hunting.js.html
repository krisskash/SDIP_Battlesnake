function getNextPosition(pos, dir) {
  switch (dir) {
    case 'up': return { x: pos.x, y: pos.y + 1 };
    case 'down': return { x: pos.x, y: pos.y - 1 };
    case 'left': return { x: pos.x - 1, y: pos.y };
    case 'right': return { x: pos.x + 1, y: pos.y };
  }
}
